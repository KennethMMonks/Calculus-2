
  	settings.render=0;

    	draw(c1,linewidth(1pt)+gray+opacity(o));

    	real x1(real t) {return t;}
		real y1(real t) {return t^2;}
		real z1(real t) {return 0;}

		path3 p1=graph(x1,y1,z1,0,1,operator ..);
